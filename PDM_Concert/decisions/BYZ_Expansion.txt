						tag = TUR

					}
				}
				relation = { who = THIS value = -50 }
			}
			any_country = {

					}
				}

			}
			any_country = {

accept_turks = {
	potential = {
		primary_culture = greek
		OR = {
			tag = BYZ
			tag = GRE
		}
		NOT = { has_country_flag = accepted_turks }
	}
	allow = {
		war = no
		is_disarmed = no
		OR = {
			government = prussian_constitutionalism
			government = hms_government
			government = democracy
		}
		nationalism_n_imperialism = 1
		
	}
	effect = {
		set_country_flag = accepted_turks
		add_accepted_culture = turkish
		any_pop = {
			limit = { has_pop_culture = turkish }
			militancy = -3
		}
	}
}

assimilate_turks = {
	potential = {
		primary_culture = greek
		OR = {
			tag = BYZ
			tag = GRE
		}
		NOT = { has_country_flag = accepted_turks }
		NOT = { has_country_flag = assimilating_turks }
	}
	allow = {
		war = no
		OR = {
				government = absolute_monarchy
				government = prussian_constitutionalism
				government = hms_government
				government = fascist_dictatorship
			}
	}
	effect = {
		any_owned = {
			limit = {
				any_pop = {
					AND = {
						is_core = TUR
						has_pop_culture = turkish
					}
				}
			}	
			add_province_modifier = {
				name = turkish_assimilation
				duration = -1
			}
		}
		consciousness = 4
		
	}
}
